1. Refactoring WAI_agent/tools/quiz_tools.py
This update replaces the heuristic grading logic with a psychometrically validated Item Response Theory (IRT) engine1. It implements response time tracking using the CHIPS methodology to detect and flag rapid guessing attempts31.






Python
import math
import time
from typing import Dict, Any, List

class EnterprisePsychometricEngine:
   """
   Implements a 4-Parameter Logistic (4PL) Item Response Theory model
   and CHIPS-based response-time validation to secure online testing [cite: 22, 31].
   """

    @staticmethod
   def calculate_item_probability(theta: float, a: float, b: float, c: float, d: float) -> float:
       """
       Calculates the probability of a correct response based on learner ability (theta)
       and calibrated item parameters: discrimination (a), difficulty (b), guessing (c), and slip (d).
       """
       try:
           # Constrain exponent to prevent floating-point overflow
           exponent = -a * (theta - b)
           exponent = max(-50.0, min(50.0, exponent))
           return c + ((d - c) / (1.0 + math.exp(exponent)))
       except (ValueError, OverflowError):
           return 0.5

    @staticmethod
   def update_learner_ability(
       current_theta: float,
       administered_items: List[Dict[str, float]],
       user_responses: List[int]
   ) -> float:
       """
       Updates the estimated learner proficiency parameter (theta) using Maximum
       Likelihood Estimation (MLE) based on active test responses [cite: 22, 30].
       """
       learning_rate = 0.5
       gradient = 0.0
       fisher_information = 0.0

       for index, item in enumerate(administered_items):
           a = item.get("discrimination", 1.0)
           b = item.get("difficulty", 0.0)
           c = item.get("guessing", 0.25)
           d = item.get("slip", 0.95)
           response = user_responses[index]  # 1 for correct, 0 for incorrect [cite: 30]

           probability = EnterprisePsychometricEngine.calculate_item_probability(current_theta, a, b, c, d)
           complement = 1.0 - probability

           if probability <= 0.0 or complement <= 0.0:
               continue

           # Calculate derivative under the 4PL model
           derivative = a * (probability - c) * (d - probability) / (d - c) if (d - c) != 0.0 else 0.1
           
           # Sum gradient and accumulated Fisher Information [cite: 30]
           gradient += (derivative / (probability * complement)) * (response - probability)
           fisher_information += (derivative ** 2) / (probability * complement)

       if fisher_information == 0.0:
           return current_theta

       # Apply update step bounded within standard psychometric limits [cite: 22, 30]
       new_theta = current_theta + (gradient / fisher_information) * learning_rate
       return max(-4.0, min(4.0, new_theta))

    @staticmethod
   def validate_response_telemetry(
       response_time: float,
       item_character_length: int,
       base_wpm: float = 200.0
   ) -> Dict[str, Any]:
       """
       Implements an Interim Person Fit (CHIPS) check to identify rapid guessing.
       """
       # Calculate the minimum time required to read and process the question text [cite: 33]
       words = max(1, item_character_length / 5.0)
       minimum_comprehension_time = (words / base_wpm) * 60.0 * 0.4  # Includes a 40% cognitive compression buffer [cite: 33]

       if response_time < minimum_comprehension_time:
           return {
               "is_valid": False,
               "anomaly_detected": "RAPID_GUESSING",
               "action": "FORCE_ROUTE_NEW_ITEM",
               "penalty_applied": True
           }

       return {
           "is_valid": True,
           "anomaly_detected": "NONE",
           "action": "PROCEED",
           "penalty_applied": False
       }

from typing import Dict, Any

class AdaptiveMetacognitiveRouter:
   """
   Evaluates assessment responses against Howell's Conscious-Competence framework
   to dynamically adjust learning paths and prevent guessing [cite: 15, 16].
   """

   COGNITIVE_STATES = {
       (True, "HIGH"): "CONSCIOUS_COMPETENCE",      # Correct & Confident
       (True, "LOW"): "UNCONSCIOUS_COMPETENCE",     # Correct & Hesitant (Possible Guess)
       (False, "LOW"): "CONSCIOUS_INCOMPETENCE",    # Incorrect & Hesitant (Aware of Gap)
       (False, "HIGH"): "UNCONSCIOUS_INCOMPETENCE"  # Incorrect & Confident (High-Risk Misconception)
   }

    @staticmethod
   def determine_adaptive_routing(
       user_id: str,
       is_correct: bool,
       confidence_level: str,
       current_path: str
   ) -> Dict[str, Any]:
       """
       Resolves the user's cognitive state and updates their learning path [cite: 1, 15].
       """
       # Normalize confidence input values
       normalized_confidence = "HIGH" if confidence_level.upper() in ["KNOW_IT", "THINK_I_KNOW_IT"] else "LOW"
       
       # Resolve the active cognitive quadrant
       cognitive_state = AdaptiveMetacognitiveRouter.COGNITIVE_STATES.get(
           (is_correct, normalized_confidence), "CONSCIOUS_INCOMPETENCE"
       )

       # Map the resolved state to an adaptive path update
       target_path = current_path
       remediation_required = False
       bypass_lockout = False

       if cognitive_state == "UNCONSCIOUS_INCOMPETENCE":
           target_path = "STANDARD_PATH"
           remediation_required = True
           bypass_lockout = True  # Immediately lock user out of fast-tracking
       elif cognitive_state == "CONSCIOUS_INCOMPETENCE":
           target_path = "STANDARD_PATH"
           remediation_required = True
       elif cognitive_state == "UNCONSCIOUS_COMPETENCE":
           # Retain on fast path but flag the concept for targeted spaced repetition review [cite: 23, 63]
           remediation_required = True
       elif cognitive_state == "CONSCIOUS_COMPETENCE":
           target_path = "FAST_TRACK"

       return {
           "user_id": user_id,
           "cognitive_state": cognitive_state,
           "updated_path": target_path,
           "remediation_triggered": remediation_required,
           "bypass_lockout": bypass_lockout,
           "routing_action": f"TRANSITION_TO_{target_path}"
       }
